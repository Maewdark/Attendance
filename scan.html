async function scanAndVerify(name) { // 1. ดึง descriptor ทั้งหมดจาก Apps Script const res = await fetch(`${APPS_SCRIPT_URL}?action=getUsers`); const users = await res.json(); // 2. สแกนหน้าปัจจุบัน const detection = await faceapi .detectSingleFace(videoEl) .withFaceLandmarks() .withFaceDescriptor(); // 3. เปรียบเทียบ const labeledDescriptors = users.map(u => new faceapi.LabeledFaceDescriptors(u.name, [new Float32Array(u.descriptor)]) ); const matcher = new faceapi.FaceMatcher(labeledDescriptors, 0.5); // threshold const match = matcher.findBestMatch(detection.descriptor); if (match.label === name && match.distance < 0.5) { // 4. บันทึกการเข้าเรียน await fetch(APPS_SCRIPT_URL, { method: "POST", body: JSON.stringify({ action: "checkIn", name }) }); alert("เช็คชื่อสำเร็จ!"); } else { alert("ไม่พบใบหน้าที่ตรงกัน"); } }
